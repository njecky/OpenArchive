```python
import sqlite3
from config.constants import *

# 🔍 CHECK LOGIN EXISTS
# ================================
def login_exists(login):

    conn = sqlite3.connect(DB_PATH)
    cursor = conn.cursor()

    cursor.execute(
        "SELECT id FROM users WHERE login = ?",
        (login,)
    )

    user = cursor.fetchone()

    conn.close()

    return user is not None


# ================================
# ➕ ADD USER
# ================================
def create_user(nom, login, mot_de_passe, role_name):

    conn = sqlite3.connect(DB_PATH)
    cursor = conn.cursor()

    try:

        # ================================
        # 🔍 GET ROLE ID
        # ================================
        cursor.execute(
            "SELECT id FROM roles WHERE name = ?",
            (role_name,)
        )

        role = cursor.fetchone()

        if not role:
            return False, "Rôle introuvable"

        role_id = role[0]

        # ================================
        # ➕ INSERT USER
        # ================================
        cursor.execute("""
            INSERT INTO users (
                nom,
                login,
                mot_de_passe,
                role_id
            )
            VALUES (?, ?, ?, ?)
        """, (
            nom,
            login,
            mot_de_passe,
            role_id
        ))

        conn.commit()

        return True, "Utilisateur ajouté"

    except sqlite3.Error as e:

        return False, str(e)

    finally:
        conn.close()
```

```python
    destination = filedialog.asksaveasfilename()

    if destination:

        shutil.copy(source_path, destination)

        # Enregistrer historique

        cursor = conn.cursor()

        cursor.execute("""

            INSERT INTO document_logs (

                user_id,
                action
            )

            VALUES (?, ?, ?)

        """, (
            document_id,
```

```python
import sqlite3
import os
import shutil

# ================================
# 📁 CREATE TABLE DOCUMENTS
# ================================
def create_documents_table():
    conn = connect_db()
    cursor = conn.cursor()

    cursor.execute("""
        CREATE TABLE IF NOT EXISTS documents (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            nom TEXT NOT NULL UNIQUE,
            type TEXT NOT NULL,
            categorie TEXT,
            tags TEXT,
            chemin TEXT NOT NULL,
            taille TEXT,
            created_at DATETIME DEFAULT CURRENT_TIMESTAMP
        )
    """)

    conn.commit()
    conn.close()


# ================================
# ➕ AJOUT DOCUMENT
# ================================
def add_document(nom, type_doc, categorie, tags, file_path):
    conn = connect_db()
    cursor = conn.cursor()

    # Vérifier doublon
    cursor.execute("SELECT id FROM documents WHERE nom = ?", (nom,))
    if cursor.fetchone():
        conn.close()
        return {"success": False, "message": "Document existe déjà"}

    # Créer dossier archive si besoin
    os.makedirs(ARCHIVE_PATH, exist_ok=True)

    # Copier fichier dans archive
    filename = os.path.basename(file_path)
    stored_path = os.path.join(ARCHIVE_PATH, filename)
    shutil.copy(file_path, stored_path)

    # Taille fichier
    size = f"{round(os.path.getsize(file_path)/1024, 2)} KB"

    cursor.execute("""
        INSERT INTO documents(
            nom, type, categorie, tags, chemin, taille
        )
        VALUES (?, ?, ?, ?, ?, ?)
    """, (
        nom,
        type_doc,
        categorie,
        tags,
        stored_path,
        size
    ))

    conn.commit()
    conn.close()

    return {"success": True, "message": "Document ajouté"}


# ================================
# 📄 GET ALL DOCUMENTS
# ================================
def get_documents():
    conn = connect_db()
    cursor = conn.cursor()

    cursor.execute("""
        SELECT id, nom, type, categorie, tags, taille, created_at
        FROM documents
        ORDER BY id DESC
    """)

    data = cursor.fetchall()
    conn.close()

    return data


# ================================
# 🔍 SEARCH DOCUMENTS
# ================================
def search_documents(keyword):
    conn = connect_db()
    cursor = conn.cursor()

    cursor.execute("""
        SELECT id, nom, type, categorie, tags, taille, created_at
        FROM documents
        WHERE nom LIKE ? OR tags LIKE ? OR categorie LIKE ?
        ORDER BY id DESC
    """, (f"%{keyword}%", f"%{keyword}%", f"%{keyword}%"))

    data = cursor.fetchall()
    conn.close()

    return data


# ================================

    conn = connect_db()
    cursor = conn.cursor()

    cursor.execute("""
        UPDATE documents

        WHERE id = ?
    """, (nom, categorie, tags, doc_id))

    conn.commit()
    conn.close()

    return {"success": True, "message": "Document mis à jour"}

    conn = connect_db()
    cursor = conn.cursor()

    # récupérer chemin fichier

        except:
            pass

    cursor.execute("DELETE FROM documents WHERE id = ?", (doc_id,))

    conn.commit()
    conn.close()

    return {"success": True, "message": "Document supprimé"}
```

```python
# ================================
# 📁 DOCUMENT MODEL - ArchivaDesk
# ================================

import sqlite3
import os
import uuid
from datetime import datetime
from config.constants import DB_PATH, DATE_FORMAT, ARCHIVE_PATH


# ================================
# 🔌 CONNEXION DB
# ================================
def connect_db():
    return sqlite3.connect(DB_PATH)


# ================================
# 📁 INITIALISATION TABLE DOCUMENTS
# ================================
def init_document_table():
    conn = connect_db()
    cursor = conn.cursor()

    cursor.execute("""
        CREATE TABLE IF NOT EXISTS documents (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            uuid TEXT UNIQUE NOT NULL,
            nom TEXT UNIQUE NOT NULL,
            type TEXT NOT NULL,
            categorie TEXT,
            tags TEXT,
            chemin TEXT NOT NULL,
            taille TEXT,
            created_at TEXT NOT NULL
        )
    """)

    conn.commit()
    conn.close()


# ================================
# 📤 AJOUT DOCUMENT
# ================================
def add_document(nom, type_file, categorie, tags, file_path, size):
    conn = connect_db()
    cursor = conn.cursor()

    # Vérifier unicité du nom
    cursor.execute("SELECT id FROM documents WHERE nom = ?", (nom,))
    if cursor.fetchone():
        conn.close()
        return {
            "success": False,
            "message": "Nom de document déjà existant"
        }

    # UUID unique
    doc_uuid = str(uuid.uuid4())

    created_at = datetime.now().strftime(DATE_FORMAT)

    # Copier fichier dans archives
    if not os.path.exists(ARCHIVE_PATH):
        os.makedirs(ARCHIVE_PATH)

    filename = os.path.basename(file_path)
    stored_path = os.path.join(ARCHIVE_PATH, f"{doc_uuid}_{filename}")

    try:
        with open(file_path, "rb") as src:
            with open(stored_path, "wb") as dst:
                dst.write(src.read())

        cursor.execute("""
            INSERT INTO documents (
                uuid, nom, type, categorie, tags, chemin, taille, created_at
            )
            VALUES (?, ?, ?, ?, ?, ?, ?, ?)
        """, (
            doc_uuid,
            nom,
            type_file,
            categorie,
            tags,
            stored_path,
            size,
            created_at
        ))

        conn.commit()

        return {
            "success": True,
            "message": "Document ajouté avec succès"
        }

    except Exception as e:
        return {
            "success": False,
            "message": str(e)
        }

    finally:
        conn.close()


# ================================
# 📥 LISTE DOCUMENTS
# ================================
def get_documents():
    conn = connect_db()
    cursor = conn.cursor()

    cursor.execute("""
        SELECT id, nom, type, categorie, tags, taille, created_at
        FROM documents
        ORDER BY id DESC
    """)

    data = cursor.fetchall()
    conn.close()

    return data


# ================================
# 👁 OUVRIR DOCUMENT
# ================================
def open_document(doc_id):
    conn = connect_db()
    cursor = conn.cursor()

    cursor.execute("SELECT chemin FROM documents WHERE id = ?", (doc_id,))
    result = cursor.fetchone()

    conn.close()

    if result:
        path = result[0]

        if os.path.exists(path):
            os.startfile(path)  # Windows
            return True

    return False


# ================================
# 🗑 SUPPRIMER DOCUMENT
# ================================
def delete_document(doc_id):
    conn = connect_db()
    cursor = conn.cursor()

    cursor.execute("SELECT chemin FROM documents WHERE id = ?", (doc_id,))
    result = cursor.fetchone()

    if not result:
        conn.close()
        return {"success": False, "message": "Document introuvable"}

    path = result[0]

    try:
        if os.path.exists(path):
            os.remove(path)

        cursor.execute("DELETE FROM documents WHERE id = ?", (doc_id,))
        conn.commit()

        return {"success": True, "message": "Document supprimé"}

    except Exception as e:
        return {"success": False, "message": str(e)}

    finally:
        conn.close()
```

Dans Tkinter, les raccourcis clavier sont **sensibles à la casse dans le nom de l’événement**, donc :

* `<Control-o>` ≠ `<Control-O>` (dans certains cas système)
* et surtout : Tkinter ne garantit pas automatiquement 
l’unification

---

# ✅ SOLUTION SIMPLE (RECOMMANDÉE)

Tu dois binder **les deux versions** :

```python id="k9d2m1"
self.root.bind("<Control-o>", lambda e: self.open_selected_document())
self.root.bind("<Control-O>", lambda e: self.open_selected_document())
```

# 🧠 SOLUTION PLUS PRO (UNE SEULE FOIS)

Tu peux aussi centraliser avec une fonction :

```python id="x8q2aa"
def bind_shortcuts(self):

    shortcuts = {
        "<Control-o>": self.open_selected_document,
        "<Control-O>": self.open_selected_document,
        "<Control-e>": self.edit_selected_document,
        "<Control-E>": self.edit_selected_document,
        "<Control-d>": self.download_selected_document,
        "<Control-D>": self.download_selected_document,
        "<Control-Delete>": self.delete_selected_document,
    }

    for key, func in shortcuts.items():
        self.root.bind(key, lambda e, f=func: f())
```

```python
import os
import platform
```

```python
# ================================
# 🖨 IMPRIMER DOCUMENT
# ================================
def print_selected_document(self):

    selected = self.document_table.selection()

    if not selected:

        messagebox.showwarning(
            "Impression",
            "Veuillez sélectionner un document."
        )

        return

    item = self.document_table.item(selected[0])

    document_name = item["values"][0]

    # récupérer chemin réel depuis DB
    conn = connect_db()

    cursor = conn.cursor()

    cursor.execute("""
        SELECT chemin
        FROM documents
        WHERE nom = ?
    """, (document_name,))

    result = cursor.fetchone()

    conn.close()

    if not result:

        messagebox.showerror(
            "Erreur",
            "Document introuvable."
        )

        return

    document_path = result[0]

    if not os.path.exists(document_path):

        messagebox.showerror(
            "Erreur",
            "Le fichier n'existe plus."
        )

        return

    try:

        # Windows
        if platform.system() == "Windows":

            os.startfile(
                document_path,
                "print"
            )

        # Linux
        elif platform.system() == "Linux":

            subprocess.run(
                ["lp", document_path]
            )

        # Mac
        elif platform.system() == "Darwin":

            subprocess.run(
                ["lp", document_path]
            )

        messagebox.showinfo(
            "Impression",
            "Document envoyé à l'imprimante."
        )

    except Exception as e:

        messagebox.showerror(
            "Erreur impression",
            str(e)
        )
```

```python
    import win32api
    WINDOWS_PRINT = True
except:
    WINDOWS_PRINT = False


# ================================
# 🖨 IMPRIMER DOCUMENT (APERÇU + ACTIONS)
# ================================
def print_selected_document(self):

    selected = self.document_table.selection()

    if not selected:
        messagebox.showwarning(
            "Impression",
            "Veuillez sélectionner un document."
        )
        return

    try:
        # ================================
        # 📄 ID DOCUMENT
        # ================================
        document_id = int(selected[0])

        document = get_document_by_id(document_id)

        if not document:
            messagebox.showerror(
                "Erreur",
                "Document introuvable."
            )
            return

        document_name = document[1]
        source_path = document[6]

        # ================================
        # ❌ FICHIER EXISTE ?
        # ================================
        if not os.path.exists(source_path):
            messagebox.showerror(
                "Erreur",
                f"Fichier introuvable :\n{source_path}"
            )
            return

        # ================================
        # 🪟 FENÊTRE D'APERÇU
        # ================================
        preview = Toplevel(self.root)
        preview.title("Aperçu du document")
        preview.geometry("600x450")
        preview.configure(bg="#ECF0F1")
        preview.grab_set()

        # ================================
        # 📌 HEADER
        # ================================
        header = Frame(preview, bg="#2C3E50", height=60)
        header.pack(fill="x")
        header.pack_propagate(False)

        Label(
            header,
            text="🖨 Aperçu du document",
            bg="#2C3E50",
            fg="white",
            font=("Helvetica", 14, "bold")
        ).pack(side="left", padx=20)

        # ================================
        # 📄 INFO DOCUMENT
        # ================================
        body = Frame(preview, bg="#ECF0F1")
        body.pack(fill="both", expand=True, padx=20, pady=20)

        Label(
            body,
            text=f"Nom : {document_name}",
            bg="#ECF0F1",
            font=("Helvetica", 11, "bold")
        ).pack(anchor="w")

        Label(
            body,
            text=f"Chemin : {source_path}",
            bg="#ECF0F1",
            font=("Helvetica", 9),
            wraplength=550,
            fg="gray"
        ).pack(anchor="w", pady=5)

        # ================================
        # 🖨 IMPRIMANTE
        # ================================
        printer_label = Label(body, text="Imprimante :", bg="#ECF0F1")
        printer_label.pack(anchor="w", pady=(15, 5))

        printers = []

        if WINDOWS_PRINT:
            printers = [p[2] for p in win32print.EnumPrinters(2)]

        printer_combo = ttk.Combobox(body, values=printers, state="readonly")
        printer_combo.pack(fill="x")
        if printers:
            printer_combo.current(0)

        # ================================
        # 🔘 FOOTER BUTTONS
        # ================================
        footer = Frame(preview, bg="#ECF0F1")
        footer.pack(fill="x", pady=10)

        # ================================
        # ❌ ANNULER
        # ================================
        def cancel():
            preview.destroy()

        Button(
            footer,
            text="Annuler",
            bg="#E74C3C",
            fg="white",
            command=cancel
        ).pack(side="right", padx=5)

        # ================================
        # ⬇ TÉLÉCHARGER
        # ================================
        def download():
            target = filedialog.asksaveasfilename(
                initialfile=document_name
            )
            if target:
                shutil.copy2(source_path, target)
                messagebox.showinfo("OK", "Téléchargé avec succès")

                    return

                if WINDOWS_PRINT:

                    win32api.ShellExecute(
                        0,
                        "print",
                        source_path,
                        f'/d:"{printer}"',
                        ".",
                        0
                    )
                else:
                    # fallback (Linux/Mac)
                    subprocess.run(["lp", "-d", printer, source_path])

                messagebox.showinfo(
                    "Succès",
                    "Document envoyé à l'imprimante"
                )

                preview.destroy()
```

```python
# ================================
# 📄 DOCUMENT VIEWER MODERNE
# ================================

import tkinter as tk
from tkinter import messagebox, simpledialog
import os

from config.constants import *
from config.function import open_file

import fitz  # PyMuPDF
from PIL import Image, ImageTk


# ================================
# 📄 VIEWER
# ================================
class DocumentViewer:

    def __init__(self, root, document_name, document_path):

        self.root = root
        self.document_name = document_name
        self.document_path = document_path

        self.zoom_level = 1.0
        self.dark_mode = False

        self.build_ui()
        self.load_document()

    # ================================
    # 🎨 UI
    # ================================
    def build_ui(self):

        self.window = tk.Toplevel(self.root)
        self.window.title(f"📄 {self.document_name}")
        self.window.geometry("1000x700")
        self.window.configure(bg=WHITE)

        # ================================
        # HEADER
        # ================================
        header = tk.Frame(self.window, bg=PRIMARY_COLOR, height=60)
        header.pack(fill="x")
        header.pack_propagate(False)

        tk.Label(
            header,
            text=self.document_name,
            bg=PRIMARY_COLOR,
            fg=WHITE,
            font=("Helvetica", 12, "bold")
        ).pack(side="left", padx=15)

        # Boutons header
        tk.Button(
            header,
            text="🔍 Rechercher",
            command=self.search_text,
            bg=SECONDARY_COLOR,
            fg=WHITE,
            bd=0
        ).pack(side="right", padx=5)

        tk.Button(
            header,
            text="➕ Zoom",
            command=self.zoom_in,
            bg=SUCCESS_COLOR,
            fg=WHITE,
            bd=0
        ).pack(side="right", padx=5)

        tk.Button(
            header,
            text="➖ Zoom",
            command=self.zoom_out,
            bg=WARNING_COLOR,
            fg=WHITE,
            bd=0
        ).pack(side="right", padx=5)

        tk.Button(
            header,
            text="🌓 Mode",
            command=self.toggle_theme,
            bg=DARK_COLOR,
            fg=WHITE,
            bd=0
        ).pack(side="right", padx=5)

        tk.Button(
            header,
            text="📂 Ouvrir externe",
            command=self.open_external,
            bg=DANGER_COLOR,
            fg=WHITE,
            bd=0
        ).pack(side="right", padx=5)

        # ================================
        # BODY
        # ================================
        self.body = tk.Frame(self.window, bg=WHITE)
        self.body.pack(fill="both", expand=True)

        # Canvas image/pdf
        self.canvas = tk.Canvas(self.body, bg=WHITE)
        self.canvas.pack(fill="both", expand=True)

        # Text preview
        self.text_area = tk.Text(
            self.body,
            bg=LIGHT_COLOR,
            fg=TEXT_COLOR,
            font=("Consolas", 10),
            wrap="word"
        )

        # Ctrl+F
        self.window.bind("<Control-f>", lambda e: self.search_text())

    # ================================
    # 📄 LOAD
    # ================================
    def load_document(self):

        if not os.path.exists(self.document_path):
            messagebox.showerror("Erreur", "Fichier introuvable")
            return

        ext = os.path.splitext(self.document_path)[1].lower()

        # ================= PDF =================
        if ext == ".pdf":
            self.show_pdf()

        # ================= IMAGE =================
        elif ext in [".png", ".jpg", ".jpeg"]:
            self.show_image()

        # ================= TEXTE =================
        elif ext in [".txt", ".csv", ".log", ".md"]:
            self.show_text()

        else:
            self.show_text()
            self.text_area.insert("end", "\n\n⚠️ Aperçu non disponible. Utilisez ouvrir.")

    # ================================
    # 📄 TEXT VIEW
    # ================================
    def show_text(self):

        self.canvas.pack_forget()
        self.text_area.pack(fill="both", expand=True)

        with open(self.document_path, "r", encoding="utf-8", errors="ignore") as f:
            content = f.read()

        self.text_area.insert("1.0", content)
        self.text_area.config(state="disabled")

    # ================================
    # 🖼 IMAGE VIEW
    # ================================
    def show_image(self):

        self.text_area.pack_forget()
        self.canvas.pack(fill="both", expand=True)

        img = Image.open(self.document_path)
        img = img.resize(
            (
                int(img.width * self.zoom_level),
                int(img.height * self.zoom_level)
            )
        )

        self.img_tk = ImageTk.PhotoImage(img)
        self.canvas.create_image(0, 0, anchor="nw", image=self.img_tk)

    # ================================
    # 📄 PDF VIEW
    # ================================
    def show_pdf(self):

        self.text_area.pack_forget()
        self.canvas.pack(fill="both", expand=True)

        doc = fitz.open(self.document_path)
        page = doc.load_page(0)

        pix = page.get_pixmap()
        img = Image.frombytes("RGB", [pix.width, pix.height], pix.samples)

        img = img.resize(
            (
                int(img.width * self.zoom_level),
                int(img.height * self.zoom_level)
            )
        )

        self.pdf_img = ImageTk.PhotoImage(img)
        self.canvas.create_image(0, 0, anchor="nw", image=self.pdf_img)

    # ================================
    # 🔍 SEARCH
    # ================================
    def search_text(self):

        if not self.text_area.winfo_ismapped():
            messagebox.showinfo("Recherche", "Recherche disponible seulement en texte.")
            return

        query = simpledialog.askstring("Recherche", "Mot à rechercher :")

        if not query:
            return

        content = self.text_area.get("1.0", "end")

        index = content.lower().find(query.lower())

        if index == -1:
            messagebox.showinfo("Recherche", "Aucun résultat trouvé")
        else:
            messagebox.showinfo("Recherche", f"Trouvé à la position {index}")

    # ================================
    # 🔍 ZOOM IN
    # ================================
    def zoom_in(self):

        self.zoom_level += 0.1
        self.load_document()

    # ================================
    # 🔍 ZOOM OUT
    # ================================
    def zoom_out(self):

        if self.zoom_level > 0.3:
            self.zoom_level -= 0.1
            self.load_document()

    # ================================
    # 🌗 THEME
    # ================================
    def toggle_theme(self):

        self.dark_mode = not self.dark_mode

        bg = DARK_COLOR if self.dark_mode else WHITE
        fg = WHITE if self.dark_mode else TEXT_COLOR

        self.window.configure(bg=bg)
        self.body.configure(bg=bg)
        self.canvas.configure(bg=bg)
        self.text_area.configure(bg=bg, fg=fg)

    # ================================
    # 📂 OPEN EXTERNAL
    # ================================
    def open_external(self):

        try:
            open_file(self.document_path)

        except Exception as e:
            messagebox.showerror("Erreur", str(e))
```

# ✅ 1. Ajouter une fonction pour récupérer l’IP locale

Ajoute ça dans ton `model.py` ou un fichier utils :

```python id="iputil1"
import socket

def get_local_ip():
    try:
        hostname = socket.gethostname()
        ip_address = socket.gethostbyname(hostname)
        return ip_address
    except Exception:
        return "unknown"
```

---

# ⚠️ Important (réseau réel)

La méthode ci-dessus donne souvent :

* ✔ IP locale (ex: 192.168.1.10)
* ❌ pas l’IP publique Internet

# ✅ 📋 GET DOCUMENT ACTIVITIES (avec recherche + filtre)

```python
# ================================
# 📋 GET DOCUMENT ACTIVITIES
# ================================
def get_document_activities(keyword="", action="Toutes"):
    """
    🔎 Récupère les activités des documents avec :
        - recherche simple (keyword)
        - filtre par action
        - combinaison des deux
        - affichage complet si aucun filtre

    💡 Algorithme utilisé :
        -> Recherche linéaire via SQL LIKE
        -> Parcours de toutes les lignes et filtrage conditionnel

    Cas supportés :
    1. 👤 Recherche simple : "contrat.pdf"
    2. 👤 Filtre action : "Consultation"
    3. 👤 Recherche + filtre : "rapport" + "Téléchargement"
    4. 👤 Aucun filtre : tout afficher
    """

    conn = connect_db()
    cursor = conn.cursor()

    # ================================
    # 🧠 BASE QUERY (toujours active)
    # ================================
    query = """
        SELECT
            id,
            document_name,
            action,
            user_id,
            ip_address,
            date_action
        FROM document_activities
        WHERE 1=1
    """

    params = []

    # ================================
    # 🔎 CAS 1 & 3 : RECHERCHE LINÉAIRE
    # ================================
    # On simule une recherche linéaire avec LIKE
    # SQLite parcourt chaque ligne et compare le texte
    if keyword and keyword.strip() != "":
        query += """
            AND document_name LIKE ?
        """
        params.append(f"%{keyword}%")

    # ================================
    # 🎯 CAS 2 & 3 : FILTRE ACTION
    # ================================
    if action and action != "Toutes":
        query += """
            AND action = ?
        """
        params.append(action)

    # ================================
    # 📊 TRI FINAL (plus récent en premier)
    # ================================
    query += " ORDER BY date_action DESC"

    # ================================
    # 🚀 EXECUTION
    # ================================
    cursor.execute(query, params)
    results = cursor.fetchall()

    conn.close()

    return results
```

---

# 🧠 Explication des 4 cas

## 👤 Cas 1 : recherche simple

```python
keyword = "contrat.pdf"
action = "Toutes"
```

👉 SQL devient :

```sql
WHERE document_name LIKE '%contrat.pdf%'
```

---

## 👤 Cas 2 : filtre action

```python
keyword = ""
action = "Consultation"
```

👉 SQL :

```sql
WHERE action = 'Consultation'
```

---

## 👤 Cas 3 : combinaison

```python
keyword = "rapport"
action = "Téléchargement"
```

👉 SQL :

```sql
WHERE document_name LIKE '%rapport%'
AND action = 'Téléchargement'
```

---

## 👤 Cas 4 : aucun filtre

```python
keyword = ""
action = "Toutes"
```

👉 SQL :

```sql
WHERE 1=1
```

➡️ affiche tout

---

# ⚡ Pourquoi c’est une "recherche linéaire" ?

Même si tu utilises SQL, SQLite fait :

* parcours de toutes les lignes
* comparaison `LIKE '%mot%'`
* filtrage ligne par ligne

👉 donc équivalent à une **Linear Search (O(n))**
